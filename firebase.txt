#include <FirebaseArduino.h>


// Set these to run the example.
#define FIREBASE_HOST "airmon8098-default-rtdb.firebaseio.com"
#define FIREBASE_AUTH "0KzXdOyTQ16gVZlRHJK2iNaTFGU0FFspBIbw1CwR"
#define WIFI_SSID "Aas"
#define WIFI_PASSWORD "17032003"


int co2;
int pm;
int ppm;
String quality;


void setup() {
  Serial.begin(115200);
  
  // Connect to Wi-Fi.
  WiFi.begin(WIFI_SSID, WIFI_PASSWORD);
  Serial.print("Connecting");
  while (WiFi.status() != WL_CONNECTED) {
    Serial.print(".");
    delay(500);
  }
  Serial.println();
  Serial.print("Connected: ");
  Serial.println(WiFi.localIP());
  
  Firebase.begin(FIREBASE_HOST, FIREBASE_AUTH);
  
  while (!Serial) {
    ; // Wait for serial port to connect. Needed for native USB port only.
  }
}


void loop() {
  while (Serial.available() == 0) {
    // Serial.write(Serial.read());
  }
  
  co2 = Serial.parseInt();
  Serial.println(co2);
  delay(1900);
  
  pm = Serial.parseInt();
  Serial.println(pm);
  delay(1400);
  
  ppm = Serial.parseInt();
  Serial.println(ppm);
  delay(900);
  
  quality = Serial.readString();
  Serial.println(quality);
  delay(400);
  
  // Set value for CO2.
  Firebase.setInt("co2", co2);
  // Handle error
  if (Firebase.failed()) {
    Serial.print("Setting /co2 failed: ");
    Serial.println(Firebase.error());
    return;
  }
  delay(1000);
  
  // Set value for ppm.
  Firebase.setInt("ppm", ppm);
  // Handle error
  if (Firebase.failed()) {
    Serial.print("Setting /ppm failed: ");
    Serial.println(Firebase.error());
    return;
  }
  delay(1000);
  
  // Set string value for quality.
  Firebase.setString("quality", quality);
  // Set value for PM.
  Firebase.setInt("PM", pm);
  // Handle error
  if (Firebase.failed()) {
    Serial.print("Setting /quality or /PM failed: ");
    Serial.println(Firebase.error());
    return;
  }
  delay(1000);

  // Serial.println(name);
  
  delay(1000);
}
